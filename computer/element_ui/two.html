<el-transfer
            v-model="value2"
            :data="data"
            :titles="['候选项', '选中']"
            @right-check-change="right_check_change">
            <span slot-scope="{ option }">
              {{ option.label }}
                <el-button
                // 此处v-if的作用是 左侧不显示这个button按钮
                v-if="flag(option)"
                type="success"
                size="mini"
                @click="change_roles(option)"
              >{{ option.roles==0?'管理员':'普通' }</el-button>
            </span>
</el-transfer>


methods:{
    // 判断item是否存在右侧选中栏  如果存在返回true 配合v-if  
    flag(e) {
      return this.value2.includes(e.key);
    },

    // 此处需求是点击按钮 更改用户权限
    // 从穿梭栏里data 找到当前项(e) 并更改某一项属性的值
    change_roles(e) {
      this.data.forEach(element => {
        if (element.key == e.key) {
          element.roles == 0 ? (element.roles = 1) : (element.roles = 0);
        }
      });
    }
}
