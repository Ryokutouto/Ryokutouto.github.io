# 可计算属性名

  • 把属性名用[ ]括起来，括号中就可以引用提前定义的变量
var prefix = 'foo'

var obj = {
  foobar: 1,
  foobaz: 2
}

obj[prefix + 'bar']  // 1
obj[prefix + 'baz']  // 2


const prefix = 'foo'

const obj = {
  [`${prefix}bar`]: 1,
  [`${prefix}baz`]: 2
}

obj.foobar     // < 1
obj['foobar']  // < 1
obj['foobaz']  // < 2

# Object.assign 浅拷贝

英 /əˈsaɪn/ 美 /əˈsaɪn/ vt. 分配；指派；[计][数] 赋值 vi. 将财产过户（尤指过户给债权人）

语法: Object.assign(target, ...sources)

参数

  • target
    • 目标对象。
  • sources
    • 源对象。

返回值

  • 目标对象。
  • 合并对象
let user = { name: "John" };

let permissions1 = { canView: true };
let permissions2 = { canEdit: true };

// 把 permissions1 和 permissions2 的所有属性都拷贝给 user
Object.assign(user, permissions1, permissions2);

// 现在 user = { name: "John", canView: true, canEdit: true }
