# 公共过滤器

  1. 创建公共函数文件(例如 filter.js )
const filter_time = function(val) {
    let time = '无'
    if (val) {
        const date = new Date(val)
        const y = date.getFullYear()
        let MM = date.getMonth() + 1
        MM = MM < 10 ? '0' + MM : MM
        let d = date.getDate()
        d = d < 10 ? '0' + d : d
        let h = date.getHours()
        h = h < 10 ? '0' + h : h
        let m = date.getMinutes()
        m = m < 10 ? '0' + m : m
        let s = date.getSeconds()
        s = s < 10 ? '0' + s : s
        time = y + '-' + MM + '-' + d + ' ' + h + ':' + m + ':' + s
    }
    return time
}

const filter_type = (val) => {
    let type = {
        '1': '中国',
        '2': '外国人',
        '3': '港澳台'
    }
    return type[val]
}
export default {
    filter_time,
    filter_type,
}
